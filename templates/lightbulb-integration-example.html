Follow these steps to integrate the light bulb animation into your existing AI CryptoDetect website:

Step 1: Include CSS & JS Files

<!-- Add to your HTML head -->
<link rel="stylesheet" href="lightbulb-component.css">

<!-- Add before closing body tag -->
<script src="lightbulb-component.js"></script>

Step 2: Add HTML Structure

<!-- Light Bulb Component HTML -->
<div class="lightbulb-animation">
  <div class="light-fixture" id="myLightBulb">
    <div class="ceiling-mount"></div>
    <div class="cord-container">
      <div class="cord"></div>
      <div class="cord-pull"></div>
    </div>
    <div class="bulb-container">
      <div class="ambient-light"></div>
      <div class="bulb on">
        <div class="reflection"></div>
      </div>
      <div class="filament"></div>
      <div class="screw-base">
        <div class="screw-threads"></div>
      </div>
    </div>
  </div>
</div>

Step 3: Initialize JavaScript

<script>
  document.addEventListener('DOMContentLoaded', function() {
    initializeLightBulb('#myLightBulb');
  });
</script>
